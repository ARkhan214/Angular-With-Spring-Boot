package com.emranhss.mkbankspring.dto;

import com.emranhss.mkbankspring.entity.TransactionType;
import java.util.Date;

public class TransactionDTO {
    private Long id;
    private TransactionType type;
    private double amount;
    private Date transactionTime;
    private String description;
    private Long accountId;          // Sender account ID
    private Long receiverAccountId;  // Receiver account ID

    public TransactionDTO() {}
    public TransactionDTO(Long id, TransactionType type, double amount, Date transactionTime,
                          String description, Long accountId, Long receiverAccountId) {
        this.id = id;
        this.type = type;
        this.amount = amount;
        this.transactionTime = transactionTime;
        this.description = description;
        this.accountId = accountId;
        this.receiverAccountId = receiverAccountId;
    }

    // getters & setters
}
